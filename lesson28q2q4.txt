    put random
end for
/* q4 */
var random2 : int
var guess : int
randint (random2, 1, 100)
put "I have generated a random number between 1 and 100. Try and guess it~"
loop
    put "What is your guess? " ..
    get guess
    if guess > random2 then
        put "That's not right, maybe you should try a smaller number?"
    elsif guess < random2 then
        put "That's not right, maybe you should try a larger number?"
    end if
    exit when random2 = guess
end loop
put "You were right! My number was ", random2, "."
